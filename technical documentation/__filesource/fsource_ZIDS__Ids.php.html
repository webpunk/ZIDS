  46.  * website developed with Zend Framework.

  56. class ZIDS_Plugin_Ids extends Zend_Controller_Plugin_Abstract 
  57. {
  58.     /**
  59.      * Contains all registered plugins
  60.      *
  61.      * @var array 
  62.      */
  63.     private $_plugins = array();
  64.  
  65.     /**
  66.      * Contains all levels and level options
  67.      *
  68.      * @var array 
  69.      */
  70.     private $_levels;
  71.     
  72.     /**
  73.      * true, if ZIDS should aggregate all impacts in the session
  74.      *
  75.      * @var boolean 
  76.      */    
  77.     private $_aggregate = false;
  78.  
  79.     /**
  80.      * contains the configurtion (usually specified in application.ini)
  81.      *
  82.      * @var array 
  83.      */    
  84.     private $_config;
  85.     
  86.     /**
  87.      * Constructor
  88.      *
  89.      * @param array|Zend_Config$config 
  90.      * @return void 
  91.      */
  92.     public function __construct(array $config) {
  93.         $this->_config = $config;
  94.  
  95.         // get all defined levels including their options
  96.         $this->_levels = $config['level'];
  97.  
  98.         // should ZIDS aggregate all impacts in the session
  99.         $this->_aggregate = $config['aggregate_in_session']; 
  100.     }
  101.  
  102.     /**
  103. /**
  104.      * Register ZIDS plugin in the pre-Dispatch phase.
  105.      * @param Zend_Controller_Request_Abstract $request 
  106.      */
  107.     public function routeShutdown(Zend_Controller_Request_Abstract $request)
  108.     {
  109.         // should ZIDS ignore this request?
  110.         foreach ($this->_config['ignore']['requests']['module'] as $i => $module) {
  111.             // if module, controller and action have been specified, all three parameters have to match
  112.             if (isset($this->_config['ignore']['requests']['controller'][$i]) && 
  113.                  isset($this->_config['ignore']['requests']['action'][$i])) {
  114.                 if ($request->getModuleName() == $module && 
  115.                     $request->getControllerName() == $this->_config['ignore']['requests']['controller'][$i] &&
  116.                     $request->getActionName() == $this->_config['ignore']['requests']['action'][$i])
  117.                         return $request;
  118.             // if only module and controller have been specified, both parameters have to match (action is being ignored)
  119.                 } else if (isset($this->_config['ignore']['requests']['controller'][$i])) {
  120.                 if ($request->getModuleName() == $module && 
  121.                     $request->getControllerName() == $this->_config['ignore']['requests']['controller'][$i])
  122.                         return $request;
  123.             // if only module has been specified, module has to match (controller & action are being ignored)
  124.                 } else if ($request->getModuleName() == $module) {
  125.                  return $request;
  126.             }
  127.         }
  128.  
  129.         // init and start PHP IDS
  130.         require_once 'IDS/Init.php';
  131.         $input = array ('REQUEST' => $_REQUEST, 
  132.                         'GET' => $_GET, 
  133.                         'POST' => $_POST, 
  134.                         'COOKIE' => $_COOKIE );
  135.         $init = IDS_Init::init ( $this->_config['phpids']['config'] );
  136.         $ids = new IDS_Monitor ( $input, $init );
  137.         $result = $ids->run ();
  138.  
  139.         // deal with the result of PHP IDS
  140.         if (! $result->isEmpty ()) {
  141.             // get PHP-IDS impact
  142.             $impact = $result->getImpact();
  143.             
  144.             // check, if ZIDS should aggregate all impacts in the session            
  145.             if ($this->_aggregate) {
  146.                 $session = new Zend_Session_Namespace('ZIDS');
  147.                 $impact += $session->impact;
  148.                 $session->impact = $impact;
  149.             }
  150.  
  151.             // find corresponding ZIDS level of attack
  152.             foreach ($this->_levels as $lvlname => $currlevel) {
  153.                 if ('*' != $lvlname)
  154.                     if (isset($currlevel['upto'])) { 
  155.                         if ($impact <= $currlevel['upto']) {
  156.                             $level = $lvlname;
  157.                             break;
  158.                         }
  159.                     } else {
  160.                         $level = $lvlname;
  161.                         break;
  162.                     }
  163.             }
  164.             if(!isset($level))
  165.                 throw new Exception('ZIDS could not find a corresponding level for impact value ' . $impact . '! Please, check your ZIDS configuration in application.ini!');
  166.                         
  167.             // which actions should ZIDS perform?
  168.             $actions = $this->_levels[$level]['action'];
  169.             // make sure to trim each action, e.g. ' email' => 'email'
  170.             array_walk($actions, create_function('&$arr','$arr=trim($arr);')); 
  171.             
  172.             // do we have to ignore this (potential) attack?
  173.             if (!in_array('ignore', $actions)) {
  174.  
  175.                 // fire all defined actions
  176.                 foreach ($actions as $action) {
  177.                     $plugin = $this->getPlugin($action);
  178.                     if (!$plugin) {
  179.                         throw new Exception('ZIDS cannot find a plugin with name ' . $action);
  180.                     }
  181.                     $plugin->injectRequest($request)
  182.                            ->fires($result, $impact, $level);
  183.                 }
  184.             }            
  185.         }
  186.         return $request;        
  187.     }
  188.     
  189.     /**
  190.      * Register a new action plugin
  191.      *
  192.      * @param ZIDS_Plugin_ActionPlugin_Interface $plugin The plugin to register
  193.      * @return ZIDS_Plugin_Ids 
  194.      */
  195.     public function registerPlugin(ZIDS_Plugin_ActionPlugin_Interface $plugin)
  196.     {
  197.         $plugin->setConfig($this->_config['level']);        
  198.         $this->_plugins[$plugin->getIdentifier()] = $plugin;
  199.         return $this;
  200.     }
  201.  
  202.     /**
  203.      * Unregister an action plugin
  204.      *
  205.      * @param string $identifier identifier of the plugin to unregister
  206.      * @return ZIDS_Plugin_Ids 
  207.      */
  208.     public function unregisterPlugin($identifier)
  209.     {
  210.         $id = strtolower($identifier);
  211.         if (isset($this->_plugins[$id])) {
  212.             unset($this->_plugins[$id]);
  213.         }
  214.         return $this;
  215.     }
  216.     
  217.     /**
  218.      * Return a registered action plugin
  219.      *
  220.      * @param string $identifier identifier of the action plugin
  221.      * @return ZIDS_Plugin_ActionPlugin_Interface or false if plugin not found
  222.      */
  223.     public function getPlugin($identifier)
  224.     {
  225.         $id = strtolower($identifier);
  226.         if (isset($this->_plugins[$id])) {
  227.             return $this->_plugins[$id];
  228.         }
  229.         return false;
  230.     }
  231. }
